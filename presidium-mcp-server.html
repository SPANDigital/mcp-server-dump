Tools

presidium_browse_collection

Navigate collections and modules within each collection

Input Schema:

{
  "type": "object",
  "properties": {
    "collection_key": {
      "type": "string",
      "description": "collection key to browse"
    },
    "module_key": {
      "type": "string",
      "description": "module key within collection"
    }
  },
  "additionalProperties": false
}

presidium_browse_module

Navigate articles within a module

Input Schema:

{
  "type": "object",
  "required": [
    "module_key"
  ],
  "properties": {
    "depth": {
      "type": "integer",
      "description": "depth of navigation"
    },
    "module_key": {
      "type": "string",
      "description": "module key to browse"
    },
    "section_id": {
      "type": "integer",
      "description": "section ID within module"
    }
  },
  "additionalProperties": false
}

presidium_get_article

Retrieve specific article by ID or path

Input Schema:

{
  "type": "object",
  "properties": {
    "article_id": {
      "type": "integer",
      "description": "article ID"
    },
    "article_path": {
      "type": "string",
      "description": "article path"
    },
    "format": {
      "type": "string",
      "description": "format for the response"
    },
    "module_key": {
      "type": "string",
      "description": "module key for article path lookup"
    }
  },
  "additionalProperties": false
}
